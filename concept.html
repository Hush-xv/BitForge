# 关键导入 — 只用了 SiliconUI 重构后的组件
from siui.components.button import SiPushButtonRefactor, SiFlatButton
from siui.components.label import SiLabelRefactor
from siui.components.container import SiBoxContainer
from siui.templates.application import SiliconApplication

# 按钮创建 — 自动获得 SiliconUI 暗色主题 + hover/click 动画
btn = SiPushButtonRefactor.withText("AND", self)
btn.setStyleData("button_color", QColor("#1f142e"))

# 显示标签 — 圆角背景 + 自定义颜色
display = SiLabelRefactor(self)
display.setBorderRadius(14)
display.setTextColor("#D1CBD4")
display.setBackgroundColor("#0d1117")

# 布局容器 — 自带弹性间距
container = SiBoxContainer(self, SiBoxContainer.TopToBottom)
container.layout().setSpacing(6)

# 动画 — 按钮按下缩放效果
self.scale_ani = SiExpAnimationRefactor(btn, "scaleFactor")
self.scale_ani.init(1/16, 0, 1, 0.9)

# 全局样式刷新
SiGlobal.siui.reloadAllWindowsStyleSheet()
